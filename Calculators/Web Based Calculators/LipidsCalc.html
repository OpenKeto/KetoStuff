<script type="text/javascript">

function write_results(){
// Set up the paragraphs for the calculated results
var outResults_parra = document.getElementById('results');
var macros_parra = document.getElementById('LipidsDiscussion');
var currentTrigs = parseFloat(document.getElementById('trigs_serum').value);
var currentHDL = parseFloat(document.getElementById('HDL_serum').value);
var currentLDL = parseFloat(document.getElementById('LDL_serum').value);
var currentFBG = parseFloat(document.getElementById('FBG_serum').value);
// Form the output string for the results
outResults_parra.innerHTML = "Triglycerides: ";
outResults_parra.innerHTML += currentTrigs + "</b>";
outResults_parra.innerHTML += " mg/dl<br/>";
outResults_parra.innerHTML += "HDL: ";
outResults_parra.innerHTML += currentHDL;
outResults_parra.innerHTML += " mg/dl<br/>";
outResults_parra.innerHTML += "LDL: ";
outResults_parra.innerHTML += currentLDL;
outResults_parra.innerHTML += " mg/dl<br/>";
outResults_parra.innerHTML += "Fasting Blood Glucose: ";
outResults_parra.innerHTML += currentFBG;
outResults_parra.innerHTML += " mg/dl<br/>";
outResults_parra.innerHTML += "<h2>Ratios and Indexes<h2/>";
var triToHdlRatio = currentTrigs / currentHDL;
outResults_parra.innerHTML += "Triglycerides/HDL Ratio: ";
outResults_parra.innerHTML += triToHdlRatio.toFixed(2);
outResults_parra.innerHTML += " mg/dl - ";
if (triToHdlRatio >= 3.8)
{
outResults_parra.innerHTML += "over 3.8 indicates a higher risk of heart disease" + "<br/>";
}
else
{
outResults_parra.innerHTML += "under 3.8 indicates a lower risk of heart disease" + "<br/>";
}
var TriGlucoseIndex = Math.log(currentTrigs * currentFBG / 2.0);
outResults_parra.innerHTML += "Triglycerides Glucose Index: ";
outResults_parra.innerHTML += TriGlucoseIndex.toFixed(3);
outResults_parra.innerHTML += " mg/dl - ";
if (TriGlucoseIndex >= 8.52)
{
outResults_parra.innerHTML += "over 8.52 indicates a higher amount of Insulin Resistance" + "<br/>";
}
else
{
outResults_parra.innerHTML += "under 8.52 indicates a lower amount of Insulin Resistance" + "<br/>";
}
}
</script>
